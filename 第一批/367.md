  const chunks = [];

  mediaRecorder.ondataavailable = (event) => {
    chunks.push(event.data);
  };

  mediaRecorder.start();
  mediaRecorder.stop();

  const blob = new Blob(chunks, { type: 'image/png' });
  const url = URL.createObjectURL(blob);
  win.loadURL(url);
});
const { ipcRenderer } = require('electron');

const { ipcMain } = require('electron');
